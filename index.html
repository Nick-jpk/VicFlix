import { useState, useEffect } from 'react';

function Counter() {
  const [count, setCount] = useState(0);
  const [timestamp, setTimestamp] = useState('');

  useEffect(() => {
    setTimestamp(new Date().toLocaleTimeString());
  }, [count]);

  return (
    <div>{count}</div>
  );
}
